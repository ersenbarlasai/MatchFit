-- 2. ŞEHİR VE İLÇE TABLOLARI
CREATE TABLE IF NOT EXISTS public.cities (
  id SERIAL PRIMARY KEY,
  name TEXT,
  lat DOUBLE PRECISION,
  lng DOUBLE PRECISION
);

CREATE TABLE IF NOT EXISTS public.districts (
  id SERIAL PRIMARY KEY,
  city_id INT REFERENCES cities(id) ON DELETE CASCADE,
  name TEXT,
  lat DOUBLE PRECISION,
  lng DOUBLE PRECISION
);

-- =========================================
-- 3. USERS SEED (MatchFit Yapısına Uygun)
-- =========================================

-- Rastgele bir batch_id belirliyoruz (Tüm veriler buna bağlanacak)
DO $$
DECLARE 
  batch_id TEXT := 'seed_matchfit_test_01';
BEGIN

  -- A. SAHTE PROFİLLERİ EKLE
  INSERT INTO public.profiles (
    id, full_name, city, district, lat, lng, 
    avatar_url, trust_score, is_seed, seed_batch_id
  )
  SELECT
    gen_random_uuid(),
    (ARRAY['Ahmet','Mehmet','Ayşe','Elif','Mert','Zeynep','Burak','Canan'])[floor(random()*8)+1] || ' ' || floor(random()*1000),
    c.name,
    d.name,
    d.lat,
    d.lng,
    'https://i.pravatar.cc/150?img=' || floor(random()*70+1)::int,
    floor(random()*40 + 60)::int, -- 60 ile 100 arası trust score
    TRUE,
    batch_id
  FROM public.cities c
  JOIN public.districts d ON d.city_id = c.id
  LIMIT 100;

  -- B. SAHTE KULLANICILARA SPOR İLGİ ALANLARI EKLE (MATCHMAKER İÇİN ŞART)
  -- Sistemdeki mevcut sporlardan (eğer sports tablonuzda veri varsa) rastgele atama yapar
  INSERT INTO public.user_sports_preferences (user_id, sport_id, skill_level)
  SELECT 
    p.id,
    s.id,
    (ARRAY['beginner', 'intermediate', 'advanced'])[floor(random()*3)+1]
  FROM public.profiles p
  CROSS JOIN LATERAL (
    SELECT id FROM public.sports ORDER BY random() LIMIT (floor(random()*3)+1)::int -- Her kullanıcıya 1-3 spor
  ) s
  WHERE p.is_seed = TRUE AND p.seed_batch_id = batch_id;

  -- C. SAHTE ETKİNLİKLER EKLE
  INSERT INTO public.events (
    id, title, city, lat, lng, start_time,
    host_id, sport_id, status, max_participants,
    is_seed, seed_batch_id
  )
  SELECT
    gen_random_uuid(),
    s.name || ' Etkinliği ' || floor(random()*100),
    p.city,
    p.lat + (random()-0.5)/10,
    p.lng + (random()-0.5)/10,
    NOW() + (random()*'5 days'::interval),
    p.id,
    s.id,
    'open',
    floor(random()*10 + 2)::int,
    TRUE,
    batch_id
  FROM public.profiles p
  -- Kullanıcının seçtiği bir sporu etkinlik için baz al
  JOIN public.user_sports_preferences usp ON usp.user_id = p.id
  JOIN public.sports s ON s.id = usp.sport_id
  WHERE p.is_seed = TRUE AND p.seed_batch_id = batch_id
  LIMIT 300;

END $$;
